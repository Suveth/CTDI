#include<iostream>
using namespace std;
int isPrime(int N){
  if(N==1)
    return 0;
  int i;
  for(i=2;i<=N/2;i++){
    if(N%i==0)
      break;
  }
  if(i>N/2)
    return 1;
  else
    return 0;
}
int main()
{
  int N,p,i;
  cin>>N;
  p=isPrime(N);
  if(p==1)
  {
    for( i=N;i!=0;i=i/10)
    {
      p=isPrime(i%10);
      if(p==0)
        break;
    }
  }
  if(i==0)
    cout<<"1";
  else
    cout<<"0";
}
