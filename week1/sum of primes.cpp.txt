class Solution{
public:
    int primeSum(int N){
        int i,sum=0,r,j;
        for(i=N;i!=0;i=i/10){
            r=i%10;
            for(j=2;j<=r/2;j++){
                if(r%j==0){
                    break;
                }
            }
        if(j>(r/2))
            sum=sum+r;
        }
        return sum;
    }
};
