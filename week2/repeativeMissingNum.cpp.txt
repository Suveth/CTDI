#include <iostream>

using namespace std;

int main()
{
    int i,j,n;
    cin>>n;
    int a[n];
    for(i=0;i<n;i++)
    cin>>a[i];
    for(i=0;i<n-1;i++){
        for(j=i+1;j<n;j++)
            if(a[i]==a[j])
                break;
        if(j<n)
            break;
    }
    cout<<a[i]<<" ";
    j=0;
    for(i=1;i<=n;i++){
      if(a[j]!=i)
        break;
    j++;
  }
  if(i<=n)
    cout<<i;

    return 0;
}
