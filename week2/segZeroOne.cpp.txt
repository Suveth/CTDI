class Solution{   
public:
   void segregate0and1(int arr[], int n) {
       int i=0,j=n-1,c =0;
       while(i<n){
           if(arr[i] == 0){
               c++;
           }
            i++;
       }
       for(i=0;i<n;i++){
           if(i<c){
               arr[i]=0;
           }
           else{
               arr[i]=1;
           }    
       }
   }
};
